# Tutor server requirements
#
# Note:
# - This project is a Node.js service, not a Python project.
# - This file documents server/runtime requirements for deployment.
# - JavaScript dependencies are managed by package.json and package-lock.json.

Node.js >= 22
npm >= 10
git

# Recommended
pm2
nginx

# Required external service
OAH API reachable via OAH_BASE_URL

# Recommended server spec for Tutor only
CPU: 2 cores
RAM: 4 GB
Disk: 40 GB
OS: Debian 12 / Ubuntu 22.04+ LTS
